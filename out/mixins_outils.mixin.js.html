/**
 * La classe outils qui permet de faire qqch
 */
class Outils {
    /**
     * Fonction permettant de renvoyer le titre d'une page en vérifiant les conditions
     * @param {string} titre Titre a passer en string
     * @returns {string} Renvoie le titre après avoir vérifié les conditions
     */
    htmlTitle(titre = null) {
        if (this.isNullOrWhiteSpace(titre)) {
            return 'LEMKA - Atelier de couture'
        } else {
            return titre + ' | LEMKA - Atelier de couture'
        }
    }

    /**
     * Fonction qui vérifie que le titre n'est pas vide ou null
     * @param {string} title le titre recu en parametre
     * @return {boolean} Retour boolean true/false
     */
    isNullOrWhiteSpace(title) {
        return title === null || title.match(/^ *$/) !== null;
    }

    /**
     * Fonction verifier que le nombre est un entier
     * @param {number} number
     * @return {boolean} retourne quelque chose je sais pas ce qu'il retourne
     */
    EstEntier(number) {
        return !isNaN(parseInt(number) && isFinite(number))
    }

    /**
     * vérifier le format, et puis le reste
     * @param {string} time valeur string a formater
     * @return {*} retourn la valeur formatée HH:MM
     */
    format(time){
        if (time) {
            return time.replace(/(?:0)?(\d+):(?:00)?(\d+).*/,'$1:$2');
        }
    }

    /**
     * Ca handleResponse
     * @param {string} response la réponse qu'il renvoie
     * @return {string|null|*} Ca retourne ce que ca doit retourner
     */
    handleResponse(response) {
        if (response.status === 204) {
            return '';
        } else if (response.status === 404) {
            return null;
        } else {
            return response.data;
        }
    }

    /**
     * on fait des toast
     * @param {string} variant la couleur bootstrap
     * @param {string} message le message a afficher dans le body
     * @param {string} title le titre a afficher dans le header
     * @param {*}vm le viem model a utiliser comme this
     */
    makeToast(variant = null, message, title, vm) {
        vm.$bvToast.toast(message, {
            title: title,
            variant: variant,
            solid: true
        })
    }

    /**
     * Fonction permettant de fermer ou ouvrir le modal
     * @param {*} vm this
     * @param {number} id Identifiant unique pour idendifier le modal
     * @param {boolean} ouvrir par default true = ouvrir le modal et false pour fermer le modal
     */
    actionModal(vm, id, ouvrir = true) {
        if (ouvrir === true) {
            vm.$bvModal.show(id)
        }
        else {
            vm.$bvModal.hide(id)
        }
    }
}

export default new Outils();
