import axios from 'axios';
import authHeader from '../auth-header';
import Outils from '@/mixins/outils.mixin'

const DOMAIN = process.env.VUE_APP_DOMAIN_API

/**
 * La classe permettant d'envoyer les requetes POST vers le serveur
 */
class UtilisateurPostService {

    /**
     * La fonction permettant de créer une mensuration spécifique lié a l'utilisateur
     * @param {*} data Objet de donnée a envoyer
     * @param {string} endpoint la route spéficique pour pour l'ajout de mensuration
     * @return {Promise<void>} retourn la réponse du serveur
     */
    postUserMensuration(data, endpoint) {
        // let endpoint = '/profil/'
        return axios
            .post(DOMAIN + endpoint, data, { headers: authHeader() })
            .then(response => { Outils.handleResponse(response) })
            .catch(error => console.log(error));
    }
}

export default new UtilisateurPostService();
